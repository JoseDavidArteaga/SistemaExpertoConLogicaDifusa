Ejemplo de Regla en Prolog 
% Regla para impacto individual alto
impacto_individual_alto(Satisfaccion, Aprendizaje, Desempeno) :-
    Satisfaccion > 80,
    Aprendizaje > 75.

impacto_individual_alto(Satisfaccion, Aprendizaje, Desempeno) :-
    Desempeno > 85,
    Satisfaccion > 70.

% Evaluación de impacto total
evaluar_impacto_total(Individual, Organizacional, muy_alto) :-
    Individual = alto,
    Organizacional = alto.
